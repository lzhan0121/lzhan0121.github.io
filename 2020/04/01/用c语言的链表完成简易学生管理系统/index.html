用c语言的链表完成简易学生管理系统

2020-04-01 · 415 话语 · 2 分钟阅读

学生管理系统

  • 使用c语言动态链表部分的知识，通过不断改变结点的方式，实现简易链表的增删改查。
#include<stdio.h>
#include<stdlib.h>
#include<malloc.h>

#define LEN sizeof(struct Student)

struct Student                                                           //定义结构体
{
    long long  num;
    float score;
    struct Student *next;
};
int n;

struct Student *creat(void)                                             //建立链表的函数
{
    struct Student *head;
    struct Student *p1, *p2;
    n = 0;
    p1 = p2 = (struct Student *)malloc(LEN);                //开辟一个新单元
              
    printf("请输入学生的学号 : ");                 //输入第一个学生的学号和成绩
    scanf("%lld",&p1->num);
    printf("请输入该学生的成绩: ");
    scanf("%f",&p1 -> score);
   
    head = NULL;
    while(p1->num!= 0)
    {
        n = n+1;
        if(n==1)head = p1;
        else p2->next = p1;
        p2 = p1;
        p1 = (struct Student *)malloc(LEN);         //开辟动态存储区，把起始地址赋给p1
        printf("\n请输入学生的学号 : ");            //输入其他学生的学号和成绩
        scanf("%lld",&p1->num);
        if(p1 -> num == 0)
        {
            system("cls");
            printf("\t\t\t\t\t\t录入结束！\n\n\n");
            printf("\t\t\t\t----------------欢迎回到学生管理系统----------------");
            printf("\n输入1新建信息\n输入2删除信息\n输入3增加信息\n输入0退出系统\n");
            break;
        }
        printf("请输入该学生的成绩: ");
        scanf("%f",&p1 -> score);
    }
    p2->next = NULL;
    return(head);
}
void print(struct Student *head)                                         //输出链表的函数
{
    int i=1;
    struct Student *p;
    printf("\n现在, 系统拥有 %d 条记录: \n",n);
    p = head;
    if(head != NULL)
    do
    {
        printf("学生%d：学号%lld 成绩：%5.1f\n", i, p->num,p->score);
        p=p->next;
        i++;
    }while(p!=NULL);
}
struct Student *del(struct Student *head, int num)                      //删除链表的函数
{
    struct Student *p1, *p2;
    if(head == NULL)
    {
        printf("error! ");
        goto END;
    }

    p1 = head;
    while(p1->num != num && p1->next != NULL)
    {
        p2 = p1;
        p1 = p1->next;
    }
    if(num == p1 -> num)
    {
        if(p1 == head)                                      //当将要删除的结点位于头结点的时候
        {
            head = p1 -> next;
        }
        else                                                         //一般情况
        {
            p2 -> next = p1 -> next;
        }
        printf("\n学号为 %lld 的学生数据删除成功\n",num);
        n = n-1;                                            //n作为全局变量，用来记录链表的数据数
    }
    else
    {
        printf("%d not been found\n",n);
    }
    END:
        return head;
}
struct Student *insert(struct Student *head, struct Student *stu)       //插入链表的函数
{
    struct Student *p0, *p1, *p2;
    p1 = head;
    p0 = stu;
    if(head == NULL)
    {
        head = p0;
        p0 -> next = NULL;
    }
    else
    {
        while(p0 -> num > p1 -> num && p1 -> next != NULL)
        {
            p2 = p1;
            p1 = p1 -> next;
        }
        if(p0 -> num <= p1 -> num)
        {
            if(head == p1)
            {
                head = p0;
            }
            else
            {
                p2 -> next = p0;
            }
            p0 -> next = p1;
        }
        else                            //p0的num最大，插入到末尾
        {
            p1 -> next = p0;
            p0 -> next = NULL;
        }
    }
    n = n+1;                         //由于插入了，增加一位成员到链表中

    return head;
}

int main()
{
    int i=1;
    long long n;
    struct Student *head, *p, stu;
    int choice;

    printf("\t\t\t\t----------------欢迎进入学生管理系统----------------");
    printf("\n输入1新建档案（输入学号时按0结束新建）\n输入2删除信息\n输入3增加信息\n输入0退出系统\n");
    while(i)
    {
        printf("\n请输入数字来操作： ");
        scanf("%d",&choice);
        while(1)
        {
            if(choice > 3)
            {
                printf("输入错误！请重新输入： ");
                scanf("%d",&choice);
            }
            else
            {
                break;
            }
            
        }
        switch(choice)                                                           //利用switch选择功能
        {
            case(0):
                system("cls");
                printf("再见！\n");
                i=0;
                break;
            case(1):
                printf("\n*********开始新建档案**********\n");
                head = creat();                                                  //调用creat函数，返回第一个结点的起始地址
                p = head;              
                print(head);
                break;                                                      //调用print函数，输出链表
            case(2):
                printf("请选择要删除的学生学号: ");
                scanf("%lld",&n);
                print(del(p,n));
                break;
            case(3):
                printf("\n请输入需要添加的学生的学号: ");
                scanf("%lld",&stu.num);
                printf("请输入需要添加的学生的分数: ");
                scanf("%f",&stu.score);
                print(insert(head,&stu));
                break;
        }
    }
    system("pause");
}
